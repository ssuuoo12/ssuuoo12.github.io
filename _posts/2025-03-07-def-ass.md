{: .message }

> 

{% highlight js %}
def cal():
    while True:
        a1 = input("첫 번째 숫자를 입력하세요(종료 q): ")
        if a1 == "q":
            break

        b1 = input("두 번째 숫자를 입력하세요(종료 q): ")
        if b1 == "q":
            break

        o = input("연산자를 선택하세요 (+, -, *, /)(종료 q): ")
        if o == "q":
            break
        try:
            a = float(a1)
            b = float(b1)
        except ValueError:
            print("숫자를 입력하세요!")
            continue

        if o == "+":
            result = a + b
        elif o == "-":
            result = a - b
        elif o == "*":
            result = a * b
        elif o == "/":
            try:
                b == 0
            except ZeroDivisionError:
                print("0으로 나눌수 없습니다")
                continue
            result = a / b
        else:
            if o not in ["+","-","*","/"]:
                print("Invalid operator")
                continue
        print(f"결과: {result}")

cal()
{% endhighlight %}
